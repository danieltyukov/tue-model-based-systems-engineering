Conveyer answers

  1. A model of the conveyer with a product can be seen in the conveyer.cif file

    conveyer.cif
    const real V_BELT = 1;      // belt velocity when moving

alg bool a_conveyer = if time <= 2: false elif time < 7: true else false end;

// position of a product on the belt, initially x = 0
cont x = 0;

equation x' = V_BELT * if a_conveyer: 1 else 0 end;
    conveyer.tooldef
    from "lib:cif" import *;

cifsim(
    "conveyer.cif",
    "-t 5",
    "-i auto",
    "--plotviz=1",
    "--plotviz-mode=live",
);

  2. A SVG drawing of the conveyer with a product is drawn in the conveyer.svg file. The needed CIF files to animate the simulation can be found in the conveyer-svg.cif and conveyer-svg.tooldef file.

    conveyer-svg.cif
    const real SVGfactor = 20;

import  "conveyer.cif";
svgfile "conveyer.svg";

svgout id "product"  attr "transform" value fmt("translate(%s,0)", x * SVGfactor);

svgout id "conveyer" attr "fill"      value if a_conveyer: "green" else "lightgrey" end;
    conveyer-svg.tooldef
    from "lib:cif" import *;

cifsim(
    "conveyer-svg.cif",
    "-i svg",
//     "--stateviz=1",
    "--frame-rate=20",
);

  3. A model of the conveyer with a toggle button can be seen in the conveyer-button.cif file. A SVG drawing of the conveyer with toggle button is drawn in the conveyer-button.svg file. The needed CIF files to animate the simulation can be found in the conveyer-button_svg.cif and conveyer-button_svg.tooldef file.

    conveyer-button.cif
    const real V_BELT = 2;

alg bool a_conveyer = a_conveyer_button.on;

cont x = 0;

equation x' = V_BELT * if a_conveyer: 1 else 0 end;

uncontrollable u_conveyer_toggle;

automaton a_conveyer_button:
    location off:
        initial;
        edge u_conveyer_toggle goto on;

    location on:
        edge u_conveyer_toggle goto off;
end
    conveyer-button-svg.cif
    const real SVGfactor = 20;

import "conveyer-button.cif";
svgfile "conveyer-button.svg";

svgout id "product"  attr "transform" value fmt("translate(%s,0)", x * SVGfactor);

svgout id "conveyer" attr "fill"      value if a_conveyer: "green" else "lightgrey" end;

svgin  id "a_conveyer"  event u_conveyer_toggle;
    conveyer-button-svg.tooldef
    from "lib:cif" import *;

cifsim(
    "conveyer-button-svg.cif",
    "-i svg",
    "--stateviz=0",
    "--frame-rate=20",
);
