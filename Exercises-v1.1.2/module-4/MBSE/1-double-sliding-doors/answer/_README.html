Double sliding doors answer

The adapted system model is provided in the files sd2-plant.cif, sd2-svg.cif, sd2-plant_io.cif, sd2-ctrl_io.cif and sd2-ctrl.cif.

sd2-plant.cif
import "sd2-plant-io.cif";

uncontrollable u_person;

automaton person:
    event reset;

    const real  X_MIN = 0;
    const real  X_MAX = 85;
    const real  V_PERSON = 20;

    cont x = X_MIN;

    location waiting:
        initial;
        equation x' = 0;
        edge u_person                               goto walking;

    location walking:
        equation x' = V_PERSON;
        edge u_person;
        edge reset      when x >= X_MAX do x := 0   goto waiting;

end

group slidingDoorLeft:
    const real X_MIN  = -45;
    const real X_MAX  = 0;
    const real V_DOOR = 30;

    cont x = X_MAX der v;

    alg real v =
                if          a_door_left and x >= X_MIN: -V_DOOR
                elif    not a_door_left and x <= X_MAX:  V_DOOR
                else                                0
                end;
end

group slidingDoorRight:
    const real X_MIN  = 0;
    const real X_MAX  = 45;
    const real V_DOOR = 30;

    cont x = X_MIN der v;

    alg real v =
                if          a_door_right and x <= X_MAX:  V_DOOR
                elif    not a_door_right and x >= X_MIN: -V_DOOR
                else                                0
                end;
end
sd2-svg.cif
import "sd2-plant-io.cif";

svgfile "sliding-doors2.svg";

// INPUTS
svgin id "person" event u_person;

// OUTPUTS
svgout id "person"              attr "display"   value if person.x >= person.X_MAX - 5: "none" else "inline" end;
svgout id "s_motion"            attr "fill"      value if s_motion: "green" else "red" end;

svgout id "person"              attr "transform" value fmt("translate(%s,0)", person.x);
svgout id "sliding_door_left"   attr "transform" value fmt("translate(%s,0)", slidingDoorLeft.x);
svgout id "sliding_door_right"  attr "transform" value fmt("translate(%s,0)", slidingDoorRight.x);
sd2-plant-io.cif
import "sd2-plant.cif";

// INPUTS
input bool a_door_left;
input bool a_door_right;


// OUTPUTS
alg bool s_motion = 30 <= person.x and person.x <= 80;
sd2-ctrl-io.cif
import "sd2-ctrl.cif";

// INPUTS
input bool s_motion;

// OUTPUTS
alg bool a_door_left    = controller.a_door_left;
alg bool a_door_right   = controller.a_door_right;
sd2-ctrl.cif
import "sd2-ctrl-io.cif";

automaton controller:
    alg bool a_door_left = open;
    alg bool a_door_right = open;

    event opening, closing;

    location closed:
        initial;
        edge opening when s_motion      goto open;

    location open:
        edge closing when not s_motion  goto closed;

end
